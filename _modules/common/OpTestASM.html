[docs]    def getcsrf(self, form):
        while True:
            try:
                myurl = urllib2.urlopen(self.url+form, timeout = 10)  
            except urllib2.URLError, e:
                time.sleep(2)
                continue
            break
        out=myurl.read()
        if 'CSRF_TOKEN' in out:
            return re.findall('CSRF_TOKEN.*value=\'(.*)\'',out)[0]
        else :
            return '0'
